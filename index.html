    const calculate = function (expression) {
        const arr = expression.split(' ');
        const stack = arr.reverse().reduce((acc, elem) => {
            if (!Number.isNaN(+elem)) {
                return [...acc, elem];
            } else {
                const operator = elem;
                const firstOperand = acc.pop();
                const secondOperand = acc.pop();
                const result = eval(`${firstOperand}${operator}${secondOperand}`);
                return [...acc, result]
            }
        }, [])
        return stack[0];
    }
    
    console.log(calculate('+ 3 4')) // 7
    console.log(calculate('- 3 * 4 5')) //-17
    console.log(calculate('* + 3 4 5')) // 35
    console.log(calculate('/ - 3 9 + 1 2')) // -2
